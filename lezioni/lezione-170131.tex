\section{Azione di $\SL_2 \left( \bbZ \right)$ su $\cH$}

Siano $F=\left\{z \in \cH | -\frac{1}{2} \leq \Re(z) \leq 0
\wedge |z| \geq 1 \right\} \cup
\left\{z \in \cH | 0 < \Re(z) < \frac{1}{2} \wedge |z| > 1 \right\}$,
$\Gamma = \SL_2 \left( \bbZ \right)/\{\pm Id\}$,   $\rho=\frac{-1+i\sqrt{3}}{2}$,
$S=\left( \begin{array}{cc} 0 & 1 \\ -1 & 0 \end{array} \right) \in \Gamma$,
$T=\left( \begin{array}{cc} 1 & 1 \\ 0 & 1 \end{array} \right) \in \Gamma$,
$\Gamma' = \langle S, T \rangle$.

\begin{teorema}
\begin{itemize}
\item[P0] $\forall z \in \cH \quad \exists g \in \Gamma$ tale che $gz \in F$
\item[P1] se $z,gz \in \bar{F}$ con $g \neq \Id$, allora $z \in \partial F$.
  PiĂč precisamente si hanno i tre casi (anche se non disgiunti):
  \begin{enumerate}
  \item[C1] $\Re z = \pm \frac{1}{2}$ e $z' = z \mp 1$;
  \item[C2] $|z| = 1$ e $z' = - \frac{1}{z}$;
  \item[C3] $z = z' = \rho$ oppure $z = z' = \rho + 1$.
  \end{enumerate}
\item[P2] se $z \in F$ e $z=gz$,
  allora $z=i$ (nel qual caso $g \in \left\langle S \right\rangle$) oppure si ha che
  $z= \rho$ (nel qual caso $g \in \left\langle ST \right\rangle$).
\end{itemize}
\end{teorema}

\begin{proof}
  \squared{P0}
Sia $\Gamma'=\left\langle S,T \right\rangle$. Sia $z \in \cH$ e cerchiamo $z' \in \Gamma'z$ con $\Im(z')$ massima possibile. Tale elemento esiste, infatti se
$\sigma = \left( \begin{array}{cc} a & b \\ c & d \end{array} \right) \in \Gamma'$, allora $\Im(\sigma z) = \frac{\Im(z)}{|cz+d|^2}$ ed essendo
$c,d \in \bbZ$, $|cz+d|^2 \geq (\Im(z))^2$ (per $c \neq 0$, se no fa $d^2 \ge 1$) per cui $\Im(\sigma z)$ Ăš limitato dall'alto e il sup viene raggiunto perchĂ©
ci sono solo un numero finito di $c,d \in \bbZ$ per cui $|cz+d|^2 \leq 1$. (Infatti siccome lo cerco massimo voglio denominatore piĂč piccolo di uno,
altrimenti giĂ  $z$ andava bene.)

\notamargine{Per vedere che ve ne sono un numero finito scriviamo $|cz+d|^2 = c^2 (\Im z)^2 + (c \Re z + d)^2 \le 1$.
  Allora ciascun termine Ăš minore di $1$, quindi ci sono un numero finito di $c$, e da questo si ricava che vi sono un numero finito di $d$.}

Possiamo quindi supporre che $z=z'$ e, a meno di comporre con qualche potenza
di $T$ (che Ăš una traslazione), supponiamo anche che $-\frac{1}{2} \leq \Re(z) < \frac{1}{2}$.

Verifichiamo quindi che $|z| \geq 1$ (se poi fosse $|z|=1$ e
$z \in \bar{F} \setminus F$ basta applicarci $S$).
$\frac{\Im(z)}{|z|^2} = \Im(Sz) \leq \Im(z)$ per la massimalitĂ  di $\Im(z)$.
Quindi $|z| \geq 1$.
\vskip 0.5cm

\squared{P1}
Siano ora $g \in \Gamma$, $z'=gz$ e supponiamo che $z,z' \in \bar{F}$ e (wlog)
che $\Im(z) \leq \Im(z')$.
$$
z'= gz \cdot 1 = \frac{az+b}{cz+d} \frac{c \bar{z} +d}{c \bar{z} +d} =
\frac{ac|z|^2 + bd + bc \bar{z} + adz}{|cz+d|^2} \stackrel{det=1}{=}
\frac{ac|z|^2 + bd + bc \bar{z} + (bc+1)z}{|cz+d|^2} =
$$
$$
\frac{ac|z|^2 + bd + bc (\bar{z} + z) + z}{|cz+d|^2} =
\frac{ac|z|^2 + bd + (2bc+1) \Re(z)}{|cz+d|^2} + i \frac{\Im(z)}{|cz+d|^2}
$$
Ora, usando che $z,z' \in \bar{F}$, per cui $|z|,|z'| \geq 1$ e
$|\Re(z)|, |\Re(z')| \leq \frac{1}{2}$, si ottiene che, se $c \neq 0$,
$\Im(z') = \frac{\Im(z)}{|cz+d|^2} \leq \frac{\Im(z)}{c^2 \Im(z)^2}
= \frac{1}{c^2 \Im(z)} \leq \frac{2}{c^2 \sqrt{3}}$ $\implies c^2 \le \frac{2}{\sqrt{3}\Im z'} \le \frac{4}{3}$
quindi $|c| \leq 1$.

\begin{itemize}
\item Se $c=0$, allora (visto che $ad - bc = 1$) deve essere $z' = z \pm b$, per cui si puĂČ avere $z'=z$ e $g=Id$; oppure
  $|\Re(z')| = \frac{1}{2}$ e $|\Re(z)| = -\frac{1}{2}$; o viceversa.
  \notamargine{Nel caso $c=0$ stando $z, z' \in F$ si ha $|\Re z - \Re z'| \le 1$}
  Negli ultimi due casi si ha $z,z' \in \partial F$ e $g=T$ e vale C1.

\item Se invece $c=1$ (il caso $c=-1$ Ăš uguale perchĂ©
  $\Gamma=\SL_2 \left( \bbZ \right)/\{\pm Id\}$),
  $\Im(z') \leq \frac{1}{\Im(z)} \stackrel{z' \in F}{\Rightarrow} \Im(z) \leq \frac{2}{\sqrt{3}}$.
  Allora $\Im(z') \leq \frac{2}{\sqrt{3}|z+d|^2} \Rightarrow
  |z+d|^2 \leq \frac{4}{3} \Rightarrow |z|^2+d^2+2d \Re(z) \leq \frac{4}{3}$. 
  Quindi (poichĂ© $\Re z \ge -\frac{1}{2}$) $d=0$ o $d=\pm 1$.

  Se $d=0$, $\Im(z') = \frac{\Im(z)}{|z|^2} \leq \Im(z)$, quindi $|z|=1$ (PerchĂ© avevamo precedentemente assunto che $\Im(z) \le \Im(z')$).
  Considerando il determinante si ottiene $b = -1$ e quindi $z' = \frac{az - 1}{z} = a - \frac{1}{z} = a - \bar{z}$. Siccome $z$ Ăš sulla circonferenza unitaria anche $- \bar{z}$ lo Ăš; visto che sia $z'$ che $z$ devono stare in $F$, deve essere $a = 0, \pm 1$.
  \begin{itemize}
  \item $a = 0$. Allora siamo nel caso C2
  \item $a = \pm 1$ allora $z = z' \in \{\rho, \rho + 1\}$ e siamo nel caso C3
  \end{itemize}

  Se $d=\pm 1$, similmente $\Im(z') = \frac{\Im(z)}{|z \pm 1|^2}$. Usando che $|z \pm 1| \ge 1$ (Fare disegno e cercare minimo modulo di $F + 1$)
  si ottiene $|z \pm 1|=1$, da cui, di nuovo, $z=\rho, z'=\rho + 1$ o viceversa e siamo nel caso C1.
\end{itemize}

\bigskip
\squared{P2} Utilizzando il punto P1 distinguiamo i tre casi:
\begin{itemize}
\item Il caso C1 non si realizza perchĂ© avevamo $z' \neq z$;
\item Il caso C2 ci da $z^2 = -1 \implies z = i$ e, per quanto visto sopra, si ha $g = S$;
\item Il caso C3 ci da solo $z = z' = \rho$ perchĂ© $\rho + 1 \notin F$ e si ottiene $g = \left(\begin{array}{cc} -1 & -1 \\ 1 & 0 \\ \end{array} \right) = (ST)^2$, quindi (visto che $(ST)^3 = \Id$) lo stabilizzatore Ăš il sottogruppo $\langle ST \rangle$.
\end{itemize}
\end{proof}

\begin{corollario}
$F$ Ăš un dominio fondamentale.

$\Stab(i)=\left\langle S \right\rangle$, 
$\Stab(\rho)=\left\langle ST \right\rangle$ e $\Stab(z)=\emptyset$ se 
$z \notin \{i, \rho \}$.
\end{corollario}

\begin{teorema}
$\Gamma$ Ăš generato da $S$ e $T$.
\end{teorema}
\notamargine{In realtĂ  si potrebbe dimostrare anche che $\Gamma$ Ăš il gruppo libero
generato da $S$ e $T$ modulo le relazioni $S^2 = Id$ e $(T S)^3=Id$.}

\begin{proof}
Vediamo ora che $\Gamma'=\Gamma$:

Sia $z=2i$ e sia $g \in \Gamma$. Per quanto visto sopra, $\exists \sigma \in \Gamma'$
tale che $\sigma g(z) \in \bar{F}$. Allora, dato che $z \notin \partial F$,
$\sigma g(z) = z$.
PoichĂ© gli unici stabilizzatori non banali sono quelli
previsti dal corollario, ne deduciamo che $\sigma g = Id$ e quindi $\Gamma = \Gamma'$.
\end{proof}

\begin{osservazione}
  Al quoziente $\cH/\Gamma \simeq F$ si puĂČ dare una struttura di superficie di Riemann
  (che non Ăš quella data dall'immersione per via dei due stabilizzatori non banali),
  identificando le rette $\Re(z)=\pm \frac{1}{2}$ e i due archi di circonferenza
  (quelli passanti per $i$) sul bordo di $F$. Il quoziente Ăš omeomorfo a $\bbC$.
  Possiamo quindi indurre una struttura di superficie di Riemann con l'omemomorfismo trovato.
\end{osservazione}

\section{Forme quadratiche binarie intere}
Una forma quadratica binaria intera Ăš un'espressione del tipo:
\[
	ax^2+bxy+cy^2 \qquad a,b,c,d \in \bbZ.
\]
Si vogliono classificare a meno di equivalenza lineare con elementi di
$\SL_2 \left( \bbZ \right)$.

\begin{osservazione}
Il discriminante $\Delta = b^2 - 4 a c$ Ăš invariante per trasformazioni lineari
invertibili. Inoltre, fissato $\Delta$, il numero di classi di equivalenza con
quel discriminante Ăš FINITO (questo perĂČ Ăš difficile).
Quando $\Delta < 0$, Ăš utile considerare $\xi \in \cH$ che risolve
$a \xi^2 + b \xi + c =0$. Per quanto abbiamo visto, $\exists \sigma \in \Gamma$
tale che $\sigma \xi \in F$. Tramite $\sigma$ si ottiene la forma ridotta secondo
Gauss.
\end{osservazione}

\section{Funzioni Modulari}
\begin{definizione}
Una funzione $f$ meromorfa su $\cH$ di dice debolmente modulare di peso $2k$ (per $k \in \mathbb{N}$), se $\forall g =
\left( \begin{array}{cc} a & b \\ c & d \end{array} \right) \in
\SL_2 \left( \bbZ \right)$ si ha:
	\[
		f(g z) = (c z + d)^{2 k} f(z)
		\qquad \forall z \in \bbC
	\]
\end{definizione}

\begin{osservazione}
	La definizione non Ăš completamente a caso. Infatti, da $a d - bc = 1$ segue $\displaystyle{\frac{d(gz)}{dz} = \frac{1}{(cz+d)^2}}$. Quindi, posto $\omega(z) = f(z)(\de z)^k$ un \emph{differenziale di peso $k$} la condizione di debole modularitĂ  si scrive come $\omega(g z) = \omega(z)$. Queste $k$-forme differenziali (secondo noi perĂČ sono tensori $(0, k)$ simmetrici).
\end{osservazione}

\begin{osservazione}
Dalla definizione segue immediatamente che $f(z+1)=f(z)$, cioĂš che una
funzione debolmente modulare Ăš periodica di periodo $1$. Definendo $q(z) = e^{2 \pi i z}$
si ha che $q$ manda $\cH$ in $D^*$ (e in particolare $\cH / \bbZ \simeq D^*$), quindi
$\widetilde{f} = f \circ q^{-1}$ Ăš meromorfa in $D^*$ e
$\displaystyle{\sum^{+\infty}_{n=-\infty}{a_n q^n}}$ Ăš la sua serie di Laurent.
Quindi $f$ si puĂČ scrivere in ``serie di Fourier'' in $q=e^{2 \pi i z}$, cioĂš
$\displaystyle{f(z)=\widetilde{f}(q)=\sum^{+\infty}_{n=-\infty}{a_n q^n}}$.
\end{osservazione}
\notamargine{Non capiamo bene come si possa dedurre la sviluppabilitĂ  in serie di Laurent
  (i poli potrebbero accumularsi in $0$). Si puĂČ invece ben fare se $f$ Ăš una funzione modulare,
  definita poco piĂč sotto.}

\begin{definizione}
Nelle notazioni di sopra, se $\widetilde{f}$ Ăš meromorfa anche in $D$, cioĂš
se gli $a_n$ con indice $n<0$ non nulli sono in numero finito, la $f$
Ăš ``meromorfa all'$\infty$'' e si dice {\bf funzione modulare}.

Se poi $\widetilde{f}$ si estende a tutto $D$, $0$ compreso, cioĂš se tutti gli $a_n$ con $n < 0$
sono nulli, la $f$ Ăš ``olomorfa all'$\infty$'' e si dice {\bf forma modulare}.

Infine, se anche $a_0 = 0$, cioĂš $f(\infty) = 0$, $f$ si dice forma modulare
cuspidale
\end{definizione}

\section{Esempio: Le Serie di Eisenstein}

\begin{definizione}
Se $L$ Ăš un reticolo in $\bbC$, per $k \geq 2$ poniamo
$G_k(L):=\displaystyle{\sum_{\lambda \in L \setminus \{ 0 \}}{\lambda ^{-2k}}}$.
\end{definizione}

\begin{osservazione}
ProprietĂ  di $G_k$:

\begin{itemize}
\item Le $G_k$ sono $(-2k)$-omogenee, cioĂš
$L_1=cL_2 \Rightarrow G_k(L_1)=c^{-2k}G_k(L_2)$.
Scrivendo $L=\bbZ \tau + \bbZ$, con $\tau \in \cH$,
si puĂČ anche vedere $G_k(L) = G_k(\tau)$ come funzione su $\cH$.
In questo modo, $\displaystyle{G_k(z)=\sum_{(m,n) \in \bbZ^2 \setminus
(0,0)}{\frac{1}{(mz+n)^{2k}}}}$, che converge uniformemente sui compatti
di $\cH$.
\item $G_k(z)$ converge puntualmente su $\cH$ e su $-\cH$, ma non su tutto $\bbC$. Infatti se $z \in \mathbb{R}$ i denominatori possono
essere arbitrariamente vicini a $0$ e quindi la serie diverge.
\item Ricordando la definizione di $g_2$ e $g_3$ si ha: $g_2(z)=60G_2(z)$ e $g_3(z)=140G_3(z)$.
\item Se $\left( \begin{array}{cc} a & b \\ c & d \end{array} \right) \in
\SL_2 \left( \bbZ \right)$,
$\displaystyle{ G_k \left(\frac{az+b}{cz+d} \right) = 
\sum_{(m,n) \in \bbZ^2 \setminus (0,0)}
{\frac{(cz+d)^{2k}} {(m(az+b)+n(cz+d))^{2k}}} =}$
$\displaystyle{ =(cz+d)^{2k} \sum_{m,n}{\frac{1} {(m(az+b)+n(cz+d))^{2k}}} =
(cz+d)^{2k} \sum_{m,n}{\frac{1}{(mz+n)^{2k}}} }$, perchĂ©
$\SL_2 \left( \bbZ \right)$ lascia invariati i reticoli. Quindi $G_k$
Ăš debolmente modulare.
\end{itemize}
\end{osservazione}


\begin{proposizione}
Le $G_k$ sono forme modulari.
\end{proposizione}

\begin{proof}
Sono tutte olomorfe su $\cH$ per teoremi classici di convergenza.
Se $G_k$ avesse un polo o un sigolaritĂ  essenziale all'$\infty$, ci sarebbero
delle successioni $\{ z_n \} \subset \cH$ tali che
$|z_n| \rightarrow +\infty$ e $|G_k(z_n)| \rightarrow +\infty$.
Ma $\displaystyle{\lim_{\Im(z) \rightarrow +\infty} G_k(z)=
2 \sum_{n=1}^{+\infty}{\frac{1}{n^{2k}}} = 2 \zeta(2k)}$, perchĂ© i termini
con $m \neq 0$ vanno a $0$ uniformemente. Quindi le $G_k$ sono olomorfe
all'$\infty$.
\end{proof}

\begin{osservazione}
$\Delta = g_2 ^3 (z) - 27 g_3 ^2 (z)$ Ăš una forma modulare di peso $12$ che
non si annulla mai in $\cH$.
\end{osservazione}
